% Korean:
% bibfile.bib의 langid 값을 기준으로 한국어 문헌을 먼저, 영어 문헌을 나중에 정렬합니다.
% 한국어 문헌은 \ktextcite, \kparencite를 사용하고 영어 문헌은 \textcite, \parencite를 사용합니다.
% 괄호형 인용 명령은 논문 템플릿과 같이 바깥 괄호를 직접 출력하지 않습니다.
%
% English:
% Sort entries by langid: Korean references first, then English references.
% Use \ktextcite and \kparencite for Korean sources; use \textcite and \parencite for English sources.
% Parenthetical citation commands do not print outer parentheses, matching the thesis template.
\DeclareSourcemap{
  \maps[datatype=bibtex]{
    \map{
      \step[fieldsource=langid, match=\regexp{korean}, final]
      \step[fieldset=presort, fieldvalue={a}]
    }
    \map{
      \step[fieldsource=langid, match=\regexp{english}, final]
      \step[fieldset=presort, fieldvalue={z}]
    }
  }
}
\DeclareDelimFormat[bib,biblist]{multinamedelim}{%
  \iffieldequalstr{langid}{korean}
  {\textperiodcentered}
  {\addcomma\space}%
}
\DeclareDelimFormat[bib,biblist]{finalnamedelim}{%
  \iffieldequalstr{langid}{korean}
  {\textperiodcentered}
  {\ifnumgreater{\value{liststop}}{2}{\finalandcomma}{}\addspace\&\space}%
}
\DeclareDelimFormat[bib,biblist]{nameyeardelim}{%
  \iffieldequalstr{langid}{korean}
  {}
  {\newunitpunct}%
}
\DeclareDelimAlias[bib,biblist]{nonameyeardelim}{nameyeardelim}

}

  \nopunct%
}

\DeclareDelimFormat[textcite]{nameyeardelim}{}

\NewDocumentCommand{\ktextcite}{o o m}{%
  \begingroup
  \UseKoreanCiteDelims
  \IfNoValueTF{#1}{%
    \textcite{#3}%
  }{%
    \IfNoValueTF{#2}{\textcite[#1]{#3}}{\textcite[#1][#2]{#3}}%
  }%
  \endgroup
}
\NewDocumentCommand{\kparencite}{o o m}{%
  \begingroup
  \UseKoreanCiteDelims
  \IfNoValueTF{#1}{%
    \cite{#3}%
  }{%

  }%
  \endgroup
}

% Korean:
% posterbox는 포스터의 각 섹션 박스입니다.
% name=은 다른 박스가 위치를 참조할 때 쓰는 내부 이름입니다.
% column=은 0, 1, 2 중 어느 단에 놓을지 정합니다.
% row=0은 해당 박스를 맨 위에서 시작하게 하고, below=다른name은 다른 박스 아래에 놓습니다.

%
% 현재 3단 배치 구조:
%   column 0: Abstract 왼쪽 부분 -> Introduction -> Figures / Tables
%   column 1: Abstract 오른쪽 부분 -> Methods -> Discussion & Conclusions
%   column 2: Poster Info -> Box Layout Guide -> Results -> References
%
% 배치 옵션을 읽는 방법:
